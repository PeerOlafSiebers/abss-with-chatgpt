model COVID19Simulation

global
{
  int totalDays <- 30;
  float initialRiskLevel <- 0.2;
  init
  {
    create SusceptibleIndividual count: 500;
    create AsymptomaticCarrier count: 50;
    create InfectedIndividual count: 10;
    create HealthcareWorker count: 50;
    create Environment count: 10;
  }
}

species SusceptibleIndividual
{
  float riskLevel <- initialRiskLevel;
  bool isInfected <- false;

  reflex assessRisk()
  {
    if (random(0, 1) < riskLevel)
    {
      become AsymptomaticCarrier();
    }
  }

  action notifyGroup(groupSize: int)
  {
    for (i in 0..groupSize)
    {
      let currentGroup = getRandomMember(getNeighbors("AsymptomaticCarrier", i));
      if (currentGroup != null && !currentGroup.isInfected)
      {
        currentGroup.become AsymptomaticCarrier();
      }
    }
  }
}

species AsymptomaticCarrier
{
  bool isInfected <- false;
  bool hasNotifiedGroup <- false;

  action updateOutbreakStatus(environment: Environment)
  {
    environment.updateOutbreakStatus();
  }

  action notifyGroup()
  {
    if (!hasNotifiedGroup)
    {
      notifyGroup(getNeighbors("SusceptibleIndividual", 10));
      hasNotifiedGroup = true;
    }
  }
}

species InfectedIndividual
{
  bool isInfected <- true;

  action notifyGroup(groupSize: int)
  {
    notifyGroup(groupSize);
  }
}

species HealthcareWorker
{
  action getPatientInfo(patient: SusceptibleIndividual | AsymptomaticCarrier | 
InfectedIndividual)
  {
    patient.getRiskLevel();
  }

  action coordinateCareWithGroup(groupSize: int)
  {
    for (i in 0..groupSize)
    {
      let currentGroup = getRandomMember(getNeighbors("SusceptibleIndividual", i));
      if (currentGroup != null && !currentGroup.isInfected)
      {
        coordinateCare(currentGroup);
      }
    }
  }

  action coordinateCare(patient: SusceptibleIndividual | AsymptomaticCarrier | 
InfectedIndividual)
  {
    patient.updateTreatmentStatus();
  }
}

species Environment
{
  float outbreakLevel <- initialRiskLevel;

  action updateOutbreakStatus()
  {
    if (random(0, 1) < outbreakLevel)
    {
      outbreakLevel += 0.1;
    }
  }
}

  output
